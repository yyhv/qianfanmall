# 一套代码实现三个版本的方案

## 概述

本文档详细说明如何通过**一套代码 + License控制**实现基础版、专业版、企业版三个版本的差异化功能。

**核心特点**：
- ✅ 同时支持管理后台和微信小程序
- ✅ License文件存储（不存数据库，防止篡改）
- ✅ RSA加密 + 数字签名（最高安全性）
- ✅ 后端统一验证（前端无法绕过）

---

## 核心思路

**关键原则**：一套代码 + License控制功能开关 = 三个版本

**实现方式**：
1. **所有功能都开发**（代码层面包含所有功能）
2. **通过License控制哪些功能可用**
3. **前端根据License动态显示/隐藏功能**
4. **后端API根据License进行权限拦截**

## 三、后端实现

### 3.1 License验证服务

```java
// LicenseService.java
@Service
public class LicenseService {

    private LicenseInfo licenseInfo;

    /**
     * 系统启动时加载License
     */
    @PostConstruct
    public void loadLicense() {
        File licenseFile = new File("license.lic");
        if (!licenseFile.exists()) {
            throw new RuntimeException("License文件不存在，系统无法启动");
        }

        // 1. 读取加密的License文件
        String encryptedLicense = FileUtil.readUtf8String(licenseFile);

        // 2. RSA解密
        String licenseJson = RSAUtil.decrypt(encryptedLicense, publicKey);

        // 3. 解析License
        JSONObject licenseObj = JSON.parseObject(licenseJson);
        JSONObject dataObj = licenseObj.getJSONObject("data");
        String signature = licenseObj.getString("signature");

        // 4. 验证数字签名（防止篡改）
        if (!RSAUtil.verify(dataObj.toJSONString(), signature, publicKey)) {
            throw new RuntimeException("License文件已被篡改，系统无法启动");
        }

        // 5. 解析License信息
        licenseInfo = dataObj.toJavaObject(LicenseInfo.class);

        // 6. 验证域名绑定
        String currentDomain = getCurrentDomain();
        if (!licenseInfo.getDomain().equals(currentDomain)) {
            throw new RuntimeException("License域名验证失败");
        }

        // 7. 验证是否过期
        if (licenseInfo.getExpireDate().before(new Date())) {
            throw new RuntimeException("License已过期，请联系供应商续费");
        }

        log.info("License验证成功：{} - {}", licenseInfo.getMerchantName(), licenseInfo.getEdition());
    }

    /**
     * 检查是否有某个模块的权限
     */
    public boolean hasModule(String moduleName) {
        return licenseInfo != null && licenseInfo.getModules().contains(moduleName);
    }

    /**
     * 获取License信息
     */
    public LicenseInfo getLicenseInfo() {
        return licenseInfo;
    }

    /**
     * 获取当前域名
     */
    private String getCurrentDomain() {
        try {
            InetAddress inetAddress = InetAddress.getLocalHost();
            return inetAddress.getHostName();
        } catch (Exception e) {
            return "localhost";
        }
    }
}
```

```java
@Aspect
@Component
public class ModulePermissionAspect {

    @Autowired
    private LicenseService licenseService;

    /**
     * 拦截带有@ModulePermission注解的方法
     */
    @Around("@within(modulePermission) || @annotation(modulePermission)")
    public Object checkModulePermission(ProceedingJoinPoint joinPoint,
                                       ModulePermission modulePermission) throws Throwable {

        String moduleName = modulePermission.value();

        // 检查License是否包含该模块
        if (!licenseService.hasModule(moduleName)) {
            throw new BusinessException("当前版本不支持此功能，请升级到更高版本");
        }

        return joinPoint.proceed();
    }
}
```

### 3.4 License信息API（管理后台和小程序共用）

```java
// LicenseController.java
@RestController
public class LicenseController {

    @Autowired
    private LicenseService licenseService;

    /**
     * 管理后台获取License信息
     */
    @GetMapping("/admin/license/info")
    public Object getAdminLicenseInfo() {
        return getLicenseInfo();
    }

    /**
     * 微信小程序获取License信息
     */
    @GetMapping("/wx/license/info")
    public Object getWxLicenseInfo() {
        return getLicenseInfo();
    }

    /**
     * 公共方法：获取License信息
     */
    private Object getLicenseInfo() {
        LicenseInfo info = licenseService.getLicenseInfo();

        Map<String, Object> result = new HashMap<>();
        result.put("edition", info.getEdition());
        result.put("modules", info.getModules());
        result.put("expireDate", info.getExpireDate());
        result.put("merchantName", info.getMerchantName());

        return ResponseUtil.ok(result);
    }

    /**
     * 上传新License（用于版本升级）
     */
    @PostMapping("/admin/license/upload")
    public Object uploadLicense(@RequestParam("file") MultipartFile file) {
        try {
            // 1. 备份旧License
            File oldLicense = new File("license.lic");
            if (oldLicense.exists()) {
                File backupLicense = new File("license_" + System.currentTimeMillis() + ".bak");
                FileUtil.copy(oldLicense, backupLicense, true);
            }

            // 2. 保存新License文件
            File licenseFile = new File("license.lic");
            file.transferTo(licenseFile);

            // 3. 重新加载License
            licenseService.loadLicense();

            // 4. 返回新的License信息
            LicenseInfo newLicense = licenseService.getLicenseInfo();

            Map<String, Object> result = new HashMap<>();
            result.put("edition", newLicense.getEdition());
            result.put("modules", newLicense.getModules());
            result.put("expireDate", newLicense.getExpireDate());
            result.put("upgradeSuccess", true);

            return ResponseUtil.ok(result);

        } catch (Exception e) {
            return ResponseUtil.fail("License上传失败：" + e.getMessage());
        }
    }
}
```

```javascript
// 过滤路由（根据License）
export function filterAsyncRoutes(routes, modules) {
  const res = []

  routes.forEach(route => {
    const tmp = { ...route }

    // 如果路由需要特定模块，检查License
    if (tmp.meta && tmp.meta.module) {
      if (!modules.includes(tmp.meta.module)) {
        return  // 没有权限，跳过此路由
      }
    }

    // 递归处理子路由
    if (tmp.children && tmp.children.length > 0) {
      tmp.children = filterAsyncRoutes(tmp.children, modules)
    }

    res.push(tmp)
  })

  return res
}
```

```javascript
const app = getApp()

function request(options) {
  return new Promise((resolve, reject) => {
    wx.request({
      url: options.url,
      method: options.method || 'GET',
      data: options.data || {},
      success: (res) => {
        // 如果返回模块权限错误，提示用户升级
        if (res.data.errno === 500 && res.data.errmsg.includes('不支持此功能')) {
          wx.showModal({
            title: '功能不可用',
            content: '当前版本不支持此功能，请联系商家升级',
            showCancel: false
          })
          reject(res.data)
        } else {
          resolve(res.data)
        }
      },
      fail: (err) => {
        reject(err)
      }
    })
  })
}

export default request
```

```java
        licenseInfo.setMerchantName(request.getMerchantName());
        licenseInfo.setDomain(request.getDomain());
        licenseInfo.setExpireDate(request.getExpireDate());
        licenseInfo.setEdition(request.getEdition());
        licenseInfo.setMaxUsers(request.getMaxUsers());

        // 2. 根据版本设置模块
        List<String> modules = getModulesByEdition(request.getEdition());
        licenseInfo.setModules(modules);

        // 3. 构建License JSON
        JSONObject dataObj = (JSONObject) JSON.toJSON(licenseInfo);
        String dataStr = dataObj.toJSONString();

        // 4. 生成数字签名（使用私钥）
        String signature = RSAUtil.sign(dataStr, PRIVATE_KEY);

        // 5. 组装License
        JSONObject licenseObj = new JSONObject();
        licenseObj.put("data", dataObj);
        licenseObj.put("signature", signature);

        // 6. RSA加密整个License
        String licenseJson = licenseObj.toJSONString();
        String encryptedLicense = RSAUtil.encrypt(licenseJson, getPublicKey());

        return encryptedLicense;
    }

    /**
     * 根据版本获取模块列表
     */
    private List<String> getModulesByEdition(String edition) {
        List<String> modules = new ArrayList<>();

        // 基础版
        modules.add("BASIC_MALL");
        modules.add("WX_MINIPROGRAM");
        modules.add("WX_SUBSCRIBE_MSG");  // 新增：小程序订阅消息

        // 专业版
        if ("PROFESSIONAL".equals(edition) || "ENTERPRISE".equals(edition)) {
            modules.add("WX_OA_TEMPLATE_MSG");  // 新增：微信服务号模板消息
            modules.add("MARKETING_COUPON");
            modules.add("MARKETING_SECKILL");
            modules.add("MARKETING_GROUPON");
            modules.add("STATISTICS");
        }

        // 企业版
        if ("ENTERPRISE".equals(edition)) {
            modules.add("MEMBER_SYSTEM");
            modules.add("DISTRIBUTION");
            modules.add("POINTS_MALL");
            modules.add("REDIS_CACHE");
        }

        return modules;
    }

    /**
     * 生成License ID
     */
    private String generateLicenseId() {
        SimpleDateFormat sdf = new SimpleDateFormat("yyyyMMdd");
        String dateStr = sdf.format(new Date());
        String randomStr = RandomStringUtils.randomNumeric(6);
        return "QF-" + dateStr + "-" + randomStr;
    }
}
```

## 八、安全机制说明

### 8.1 为什么前端控制不安全？

**小程序代码可以被反编译**：

1. **获取小程序包**：使用工具提取wxapkg文件
2. **反编译**：得到JavaScript源码
3. **修改代码**：删除权限检查逻辑
4. **重新打包**：生成新的小程序包

**但是！即使前端被破解，后端仍然安全**：

```java
// 攻击者修改小程序代码，删除了权限检查
// 但是当小程序调用API时，后端会拦截：

@GetMapping("/wx/seckill/list")
@ModulePermission("MARKETING_SECKILL")  // 后端强制验证
public Object list() {
    // 如果没有MARKETING_SECKILL权限
    // AOP拦截器会抛出异常：BusinessException("当前版本不支持此功能")
    return seckillService.list();
}
```

**结论**：
- ✅ **前端控制**：提升用户体验，避免显示无权限的功能
- ✅ **后端控制**：真正的安全保障，无法绕过

### 8.2 License文件防篡改机制

1. **数字签名**：使用RSA私钥对License数据签名
2. **RSA加密**：整个License文件使用RSA加密
3. **验证流程**：
   - 解密License文件
   - 提取数据和签名
   - 使用公钥验证签名
   - 如果数据被修改，签名验证失败

**商家无法修改License的原因**：
- ❌ 没有私钥，无法生成有效签名
- ❌ 修改数据后，签名验证失败
- ❌ 无法破解RSA加密

```java
import com.lyqf.license.exception.LicenseException;
import com.lyqf.license.model.LicenseInfo;
import com.lyqf.license.util.RSAUtil;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;
import org.springframework.beans.factory.annotation.Value;
import org.springframework.stereotype.Service;

import javax.annotation.PostConstruct;
import java.io.File;
import java.net.InetAddress;
import java.util.Date;

/**
 * License验证服务
 */
@Service
public class LicenseService {

    private static final Logger log = LoggerFactory.getLogger(LicenseService.class);

    private LicenseInfo licenseInfo;

    @Value("${license.file.path:license.lic}")
    private String licenseFilePath;

    @Value("${license.public.key:}")
    private String publicKey;

    /**
     * 系统启动时加载License
     */
    @PostConstruct
    public void loadLicense() {
        File licenseFile = new File(licenseFilePath);
        if (!licenseFile.exists()) {
            throw new LicenseException("License文件不存在，系统无法启动");
        }

        try {
            // 1. 读取加密的License文件
            String encryptedLicense = FileUtil.readUtf8String(licenseFile);

            // 2. RSA解密
            String licenseJson = RSAUtil.decrypt(encryptedLicense, publicKey);

            // 3. 解析License
            JSONObject licenseObj = JSON.parseObject(licenseJson);
            JSONObject dataObj = licenseObj.getJSONObject("data");
            String signature = licenseObj.getString("signature");

            // 4. 验证数字签名（防止篡改）
            if (!RSAUtil.verify(dataObj.toJSONString(), signature, publicKey)) {
                throw new LicenseException("License文件已被篡改，系统无法启动");
            }

            // 5. 解析License信息
            licenseInfo = dataObj.toJavaObject(LicenseInfo.class);

            // 6. 验证域名绑定
            String currentDomain = getCurrentDomain();
            if (!licenseInfo.getDomain().equals(currentDomain)) {
                throw new LicenseException("License域名验证失败");
            }

            // 7. 验证是否过期
            if (licenseInfo.getExpireDate().before(new Date())) {
                throw new LicenseException("License已过期，请联系供应商续费");
            }

            log.info("License验证成功：{} - {}", licenseInfo.getMerchantName(), licenseInfo.getEdition());

        } catch (Exception e) {
            log.error("License验证失败", e);
            throw new LicenseException("License验证失败：" + e.getMessage());
        }
    }

    /**
     * 检查是否有某个模块的权限
     */
    public boolean hasModule(String moduleName) {
        return licenseInfo != null && licenseInfo.getModules().contains(moduleName);
    }

    /**
     * 获取License信息
     */
    public LicenseInfo getLicenseInfo() {
        return licenseInfo;
    }

    /**
     * 获取当前域名
     */
    private String getCurrentDomain() {
        try {
            InetAddress inetAddress = InetAddress.getLocalHost();
            return inetAddress.getHostName();
        } catch (Exception e) {
            return "localhost";
        }
    }
}
```

#### 13.4.6 RSAUtil.java

```java
package com.lyqf.license.util;

import javax.crypto.Cipher;
import java.security.*;
import java.security.interfaces.RSAPrivateKey;
import java.security.interfaces.RSAPublicKey;
import java.security.spec.PKCS8EncodedKeySpec;
import java.security.spec.X509EncodedKeySpec;
import java.util.Base64;

/**
 * RSA工具类
 */
public class RSAUtil {

    /**
     * 生成RSA密钥对
     */
    public static void generateKeyPair() throws Exception {
        KeyPairGenerator keyPairGenerator = KeyPairGenerator.getInstance("RSA");
        keyPairGenerator.initialize(2048);
        KeyPair keyPair = keyPairGenerator.generateKeyPair();

        RSAPublicKey publicKey = (RSAPublicKey) keyPair.getPublic();
        RSAPrivateKey privateKey = (RSAPrivateKey) keyPair.getPrivate();

        String publicKeyStr = Base64.getEncoder().encodeToString(publicKey.getEncoded());
        String privateKeyStr = Base64.getEncoder().encodeToString(privateKey.getEncoded());

        System.out.println("公钥：" + publicKeyStr);
        System.out.println("私钥：" + privateKeyStr);
    }

    /**
     * 加密
     */
    public static String encrypt(String data, String publicKey) throws Exception {
        byte[] keyBytes = Base64.getDecoder().decode(publicKey);
        X509EncodedKeySpec keySpec = new X509EncodedKeySpec(keyBytes);
        KeyFactory keyFactory = KeyFactory.getInstance("RSA");
        PublicKey pubKey = keyFactory.generatePublic(keySpec);

        Cipher cipher = Cipher.getInstance("RSA");
        cipher.init(Cipher.ENCRYPT_MODE, pubKey);
        byte[] encryptedBytes = cipher.doFinal(data.getBytes());

        return Base64.getEncoder().encodeToString(encryptedBytes);
    }

    /**
     * 解密
     */
    public static String decrypt(String encryptedData, String privateKey) throws Exception {
        byte[] keyBytes = Base64.getDecoder().decode(privateKey);
        PKCS8EncodedKeySpec keySpec = new PKCS8EncodedKeySpec(keyBytes);
        KeyFactory keyFactory = KeyFactory.getInstance("RSA");
        PrivateKey privKey = keyFactory.generatePrivate(keySpec);

        Cipher cipher = Cipher.getInstance("RSA");
        cipher.init(Cipher.DECRYPT_MODE, privKey);
        byte[] decryptedBytes = cipher.doFinal(Base64.getDecoder().decode(encryptedData));

        return new String(decryptedBytes);
    }

    /**
     * 签名
     */
    public static String sign(String data, String privateKey) throws Exception {
        byte[] keyBytes = Base64.getDecoder().decode(privateKey);
        PKCS8EncodedKeySpec keySpec = new PKCS8EncodedKeySpec(keyBytes);
        KeyFactory keyFactory = KeyFactory.getInstance("RSA");
        PrivateKey privKey = keyFactory.generatePrivate(keySpec);

        Signature signature = Signature.getInstance("SHA256withRSA");
        signature.initSign(privKey);
        signature.update(data.getBytes());

        return Base64.getEncoder().encodeToString(signature.sign());
    }

    /**
     * 验证签名
     */
    public static boolean verify(String data, String sign, String publicKey) throws Exception {
        byte[] keyBytes = Base64.getDecoder().decode(publicKey);
        X509EncodedKeySpec keySpec = new X509EncodedKeySpec(keyBytes);
        KeyFactory keyFactory = KeyFactory.getInstance("RSA");
        PublicKey pubKey = keyFactory.generatePublic(keySpec);

        Signature signature = Signature.getInstance("SHA256withRSA");
        signature.initVerify(pubKey);
        signature.update(data.getBytes());

        return signature.verify(Base64.getDecoder().decode(sign));
    }
}
```

## 十五、微信消息通知功能

### 15.1 功能概述

本系统提供两种微信消息通知方式：

| 消息类型 | 适用场景 | 版本要求 | 配置复杂度 |
|---------|---------|---------|-----------|
| **小程序订阅消息** | 小程序内一次性通知，如订单状态变更 | 基础版及以上 | 简单 |
| **微信服务号模板消息** | 服务号内通知，支持更多场景，无需用户每次授权 | 专业版及以上 | 中等 |

### 15.2 两种消息方式的对比

#### 15.2.1 小程序订阅消息

**特点**：
- 用户需要主动点击订阅按钮授权
- 每次订阅只能发送一次消息（一次性订阅）
- 支持长期订阅（需微信审核，仅特定行业）
- 消息入口在小程序内

**适用场景**：
| 场景 | 说明 | 推荐指数 |
|-----|------|---------|
| 订单支付成功 | 用户下单后通知 | ⭐⭐⭐⭐⭐ |
| 发货通知 | 商家发货后通知用户 | ⭐⭐⭐⭐⭐ |
| 订单完成 | 订单确认收货 | ⭐⭐⭐⭐ |
| 退款通知 | 退款进度通知 | ⭐⭐⭐⭐ |
| 秒杀提醒 | 秒杀开始前提醒（需用户主动订阅） | ⭐⭐⭐ |

**使用时机**：
- 用户在小程序内完成某个操作后，主动订阅相关通知
- 例如：用户下单成功后，弹出订阅"发货通知"的弹窗

#### 15.2.2 微信服务号模板消息

**特点**：
- 用户关注服务号后即可接收，无需每次授权
- 可以主动推送给用户
- 消息入口在微信聊天列表，更显眼
- 支持更多业务场景

**适用场景**：
| 场景 | 说明 | 推荐指数 |
|-----|------|---------|
| 营销活动通知 | 促销活动、优惠券发放 | ⭐⭐⭐⭐⭐ |
| 会员等级变更 | 会员升级/降级通知 | ⭐⭐⭐⭐⭐ |
| 分销收益通知 | 分销佣金到账 | ⭐⭐⭐⭐⭐ |
| 积分变动通知 | 积分获取/消耗 | ⭐⭐⭐⭐ |
| 订单状态变更 | 全流程订单跟踪 | ⭐⭐⭐⭐ |
| 秒杀开始提醒 | 营销活动提醒 | ⭐⭐⭐⭐ |
| 拼团成功/失败 | 拼团结果通知 | ⭐⭐⭐⭐ |

**使用时机**：
- 需要主动触达用户的营销场景
- 需要更强的消息提醒效果
- 用户可能长时间未打开小程序时

### 15.3 消息发送策略

#### 15.3.1 自动选择最优通道

系统应根据场景自动选择消息通道：

```
              ┌──────────────┴──────────────┐

     ┌─────────────────┐          ┌─────────────────┐
     │ 订单相关通知     │          │ 营销/会员通知   │

     └────────┬────────┘          └────────┬────────┘
              │                             │

              │                             │
              │                    ┌───────┴───────┐
              │                    ▼               ▼
              │           ┌─────────────┐ ┌─────────────┐
```

#### 15.3.2 消息发送优先级

1. **订单支付成功**：小程序订阅消息（必发）
2. **订单发货通知**：小程序订阅消息 + 服务号模板消息（双通道）

```java
// WxOaConfigService.java
@Service
public class WxOaConfigService {

    @Autowired
    private SystemConfigService systemConfigService;

    @Autowired
    private LicenseService licenseService;

    /**
     * 检查服务号功能是否可用
     * 需要同时满足两个条件：
     * 1. License包含WX_OA_TEMPLATE_MSG模块
     * 2. 商家已开启服务号功能
     */
    public boolean isWxOaEnabled() {
        // 1. 检查License权限
        if (!licenseService.hasModule("WX_OA_TEMPLATE_MSG")) {
            return false;
        }

        // 2. 检查商家是否开启
        String enabled = systemConfigService.getConfigValue("wx_oa_enabled");
        return "true".equals(enabled);
    }

    /**
     * 获取服务号配置
     */
    public WxOaConfig getWxOaConfig() {
        if (!isWxOaEnabled()) {
            return null;
        }

        WxOaConfig config = new WxOaConfig();
        config.setAppId(systemConfigService.getConfigValue("wx_oa_appid"));
        config.setSecret(systemConfigService.getConfigValue("wx_oa_secret"));
        return config;
    }

    /**
     * 更新服务号配置
     */
    public void updateWxOaConfig(WxOaConfig config) {
        // 检查License权限
        if (!licenseService.hasModule("WX_OA_TEMPLATE_MSG")) {
            throw new BusinessException("当前版本不支持微信服务号功能，请升级到专业版");
        }

        systemConfigService.updateConfig("wx_oa_enabled", String.valueOf(config.isEnabled()));
        systemConfigService.updateConfig("wx_oa_appid", config.getAppId());
        systemConfigService.updateConfig("wx_oa_secret", config.getSecret());
    }
}
```

```java
            } catch (Exception e) {
                log.error("订单支付通知-小程序消息发送失败", e);
            }

        if (wxOaConfigService.isWxOaEnabled() && user.getOaOpenId() != null) {
            try {

            } catch (Exception e) {
                log.error("订单支付通知-服务号消息发送失败", e);
            }
        }
    }

    public void sendOrderShipNotice(Order order, User user) {
        boolean miniprogramSent = false;
        boolean oaSent = false;

        // 1. 发送小程序订阅消息

            try {
                wxOaTemplateMsgService.sendOrderShipMsg(order, user);
                oaSent = true;
                log.info("发货通知-服务号消息发送成功，订单号：{}", order.getOrderSn());
            } catch (Exception e) {
                log.error("发货通知-服务号消息发送失败", e);
            }

        if (!miniprogramSent && !oaSent) {
            log.warn("发货通知发送失败，订单号：{}，所有通道均发送失败", order.getOrderSn());
```

```javascript
/**
 * 请求订阅消息授权
 * @param {Array} types 订阅类型数组
 */
function requestSubscribe(types) {
  return new Promise((resolve, reject) => {
    const templateIds = types.map(type => subscribeTypes[type].templateId)

    wx.requestSubscribeMessage({
      tmplIds: templateIds,
      success: (res) => {
        // 记录用户订阅结果
        const subscribedTypes = []
        types.forEach(type => {
          const templateId = subscribeTypes[type].templateId
          if (res[templateId] === 'accept') {
            subscribedTypes.push(type)
          }
        })

        if (subscribedTypes.length > 0) {
          // 同步到服务器
          syncSubscribeToServer(subscribedTypes)
        }

        resolve(subscribedTypes)
      },
      fail: (err) => {
        console.error('订阅消息授权失败', err)
        reject(err)
      }
    })
  })
}

/**
 * 同步订阅记录到服务器
 */
function syncSubscribeToServer(types) {
  wx.request({
    url: 'https://api.example.com/wx/subscribe/sync',
    method: 'POST',
    data: { types },
    success: () => {
      console.log('订阅记录同步成功')
    }
  })
}
```

```java
     * 发货 - 上传快递单号

    @Transactional
    public void ship(OrderShipDTO shipDTO) {
        // 1. 查询订单
        Order order = orderMapper.selectByPrimaryKey(shipDTO.getOrderId());
        if (order == null) {
            throw new BusinessException("订单不存在");
        }

        if (!order.getOrderStatus().equals(OrderStatus.SHIP_STATUS)) {
            throw new BusinessException("订单状态不允许发货");
        }

        // 2. 保存物流信息
        OrderExpress orderExpress = orderExpressMapper.selectByOrderId(order.getId());
        if (orderExpress == null) {
            orderExpress = new OrderExpress();
            orderExpress.setOrderId(order.getId());
            orderExpress.setOrderSn(order.getOrderSn());
            orderExpress.setMobile(order.getMobile());
        }

        orderExpress.setExpressCompany(shipDTO.getExpressCompany());
        orderExpress.setExpressCompanyName(shipDTO.getExpressCompanyName());
        orderExpress.setExpressNo(shipDTO.getExpressNo());
        orderExpress.setShipTime(new Date());
        orderExpress.setExpressStatus("SHIPPED");
        orderExpress.setUpdateTime(new Date());

        if (orderExpress.getId() == null) {
            orderExpress.setAddTime(new Date());
            orderExpressMapper.insertSelective(orderExpress);
        } else {
            orderExpressMapper.updateByPrimaryKeySelective(orderExpress);
        }

        // 3. 更新订单状态
        order.setOrderStatus(OrderStatus.SHIP_STATUS);
        order.setShipSn(shipDTO.getExpressNo());
        order.setShipChannel(shipDTO.getExpressCompanyName());
        order.setShipTime(new Date());
        order.setUpdateTime(new Date());
        orderMapper.updateByPrimaryKeySelective(order);

        // 4. 发送发货通知
        User user = userMapper.selectByPrimaryKey(order.getUserId());
        messageSendService.sendOrderShipNotice(order, user);

        log.info("订单发货成功，订单号：{}，快递单号：{}", order.getOrderSn(), shipDTO.getExpressNo());
    }

    /**
     * 根据手机号查询物流信息

    public List<ExpressTraceVO> queryByMobile(String mobile) {
        // 1. 查询该手机号关联的所有订单物流

        if (expressList == null || expressList.isEmpty()) {

        // 2. 组装返回数据

            vo.setExpressStatus(express.getExpressStatus());

            // 查询物流轨迹
            List<ExpressTrace> traces = expressTraceMapper.selectByExpressNo(express.getExpressNo());
            vo.setTraces(traces);

            result.add(vo);
        }

        return result;
    }

            try {

                if (traces != null && !traces.isEmpty()) {
                    // 保存物流轨迹

                        ExpressTrace trace = new ExpressTrace();
                        trace.setOrderId(express.getOrderId());
                        trace.setExpressNo(express.getExpressNo());
                        trace.setTraceTime(dto.getTime());
                        trace.setTraceInfo(dto.getInfo());
                        trace.setTraceLocation(dto.getLocation());
                        trace.setAddTime(new Date());
                        expressTraceMapper.insertSelective(trace);
                    }

                    // 检查是否已签收
```

```java
    /**
     * 批量发货
     */
    @PostMapping("/batchShip")
    public Object batchShip(@RequestBody BatchShipDTO batchShipDTO) {
        if (batchShipDTO.getOrderIds() == null || batchShipDTO.getOrderIds().isEmpty()) {
            return ResponseUtil.fail("请选择要发货的订单");
        }
        if (StringUtils.isEmpty(batchShipDTO.getExpressCompany())) {
            return ResponseUtil.fail("请选择物流公司");
        }

        int successCount = 0;
        int failCount = 0;
        List<String> failMessages = new ArrayList<>();

        for (Integer orderId : batchShipDTO.getOrderIds()) {
            try {
                OrderShipDTO shipDTO = new OrderShipDTO();
                shipDTO.setOrderId(orderId);
                shipDTO.setExpressCompany(batchShipDTO.getExpressCompany());
                shipDTO.setExpressCompanyName(batchShipDTO.getExpressCompanyName());
                shipDTO.setExpressNo(batchShipDTO.getExpressNoPrefix() + orderId); // 自动生成快递单号

                expressService.ship(shipDTO);
                successCount++;
            } catch (Exception e) {
                failCount++;
                failMessages.add("订单" + orderId + "发货失败：" + e.getMessage());
            }
        }

        Map<String, Object> result = new HashMap<>();
        result.put("successCount", successCount);
        result.put("failCount", failCount);
        result.put("failMessages", failMessages);

        return ResponseUtil.ok(result);
    }
}
```

```java
// WxExpressController.java
@RestController
@RequestMapping("/wx/express")
public class WxExpressController {

    @Autowired
    private ExpressService expressService;

    @Autowired
    private OrderService orderService;

    /**
     * 根据订单ID查询物流信息
     */
    @GetMapping("/order/{orderId}")
    public Object queryByOrderId(@LoginUser Integer userId, @PathVariable Integer orderId) {
        // 验证订单归属
        Order order = orderService.findById(orderId);
        if (order == null) {
            return ResponseUtil.fail("订单不存在");
        }
        if (!order.getUserId().equals(userId)) {
            return ResponseUtil.fail("无权查看该订单");
        }

        // 查询物流信息
        ExpressTraceVO express = expressService.queryByOrderId(orderId);
        if (express == null) {
            return ResponseUtil.fail("暂无物流信息");
        }

        return ResponseUtil.ok(express);
    }

    /**
     * 根据手机号查询物流信息
     * 无需登录，方便用户快速查询
     */
    @GetMapping("/mobile")
    public Object queryByMobile(@RequestParam String mobile) {
        // 手机号格式校验
        if (!mobile.matches("^1[3-9]\\d{9}$")) {
            return ResponseUtil.fail("手机号格式不正确");
        }

        // 查询物流信息
        List<ExpressTraceVO> expressList = expressService.queryByMobile(mobile);
        if (expressList == null || expressList.isEmpty()) {
            return ResponseUtil.fail("未查询到物流信息");
        }

        return ResponseUtil.okList(expressList);
    }
}
```
